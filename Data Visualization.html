# --- PLOT B: BAR PLOT (Distribution across Categories) ---
# Create an artificial categorical column for data categorization
df['TAX_LEVEL'] = pd.qcut(df['TAX'], q=3, labels=['Low Tax', 'Medium Tax', 'High Tax'])
plt.figure(figsize=(8, 5))
sns.barplot(data=df, x='TAX_LEVEL', y='MEDV', estimator='mean', palette='muted', hue='TAX_LEVEL', legend=False)
plt.title('Average House Price by Property Tax Bracket', fontsize=14, fontweight='bold', pad=15)
plt.xlabel('Tax Bracket', fontsize=12)
plt.ylabel('Average Price ($1000s)', fontsize=12)
plt.tight_layout()
# Export plot as image
plt.savefig('bar_tax_vs_price.png', dpi=300)
plt.show()
