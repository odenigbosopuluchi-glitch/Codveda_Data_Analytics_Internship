# ---------------------------------------------------------
# STEP 2: Setup Classification Framework
# ---------------------------------------------------------
# Extract the final numerical column as our regression house price target
raw_target_col = df.columns[-1]
X_raw = df.drop(columns=[raw_target_col])

# Convert continuous price values to binary category classes per assignment rules:
# 1 = High Price (Above median), 0 = Low Price (Below median)
median_price = df[raw_target_col].median()
y = (df[raw_target_col] > median_price).astype(int)

# Automatically identify feature datatypes
NUMERIC_FEATURES = X_raw.select_dtypes(include=[np.number]).columns.tolist()
CATEGORICAL_FEATURES = X_raw.select_dtypes(exclude=[np.number]).columns.tolist()

# ---------------------------------------------------------
# STEP 3: Preprocessing Setup & Train/Test Split
# ---------------------------------------------------------
preprocessor = ColumnTransformer(
    transformers=[
        ('num', StandardScaler(), NUMERIC_FEATURES),
        ('cat', OneHotEncoder(drop='first', handle_unknown='ignore'), CATEGORICAL_FEATURES)
    ]
)

# Splitting dataset into training and testing partitions
X_train, X_test, y_train, y_test = train_test_split(X_raw, y, test_size=0.2, random_state=42, stratify=y)

print(f"Training features shape: {X_train.shape}")
print(f"Testing features shape: {X_test.shape}")

# ---------------------------------------------------------
# STEP 4: Train and Evaluate Multiple Base Classification Models
# ---------------------------------------------------------
models = {
    'Logistic Regression': LogisticRegression(random_state=42, max_iter=1000),
    'Decision Tree': DecisionTreeClassifier(random_state=42),
    'Random Forest': RandomForestClassifier(random_state=42)
}

results = {}

print("\n--- Base Model Evaluation ---")
for name, model in models.items():
    pipeline = Pipeline(steps=[('preprocessor', preprocessor), ('classifier', model)])
    pipeline.fit(X_train, y_train)
    y_pred = pipeline.predict(X_test)
    
    results[name] = {
        'Accuracy': accuracy_score(y_test, y_pred),
        'Precision': precision_score(y_test, y_pred, zero_division=0),
        'Recall': recall_score(y_test, y_pred, zero_division=0),
        'F1-Score': f1_score(y_test, y_pred, zero_division=0)
    }
    
    print(f"\n[{name}]")
    for metric, value in results[name].items():
        print(f"  {metric}: {value:.4f}")

# ---------------------------------------------------------
# STEP 5 & 6: Fixed Hyperparameter Tuning & Plotting
# ---------------------------------------------------------
print("\n--- Tuning Random Forest via Grid Search ---")

# 1. Define lists using plain variables so they don't get cut out
list_of_trees = [50, 100, 150]
list_of_splits = [2, 5]
list_of_depths = [None, 5, 10]

# 2. Setup the parameter dictionary cleanly
param_grid = {
    'classifier__n_estimators': list_of_trees,
    'classifier__max_depth': list_of_depths,
    'classifier__min_samples_split': list_of_splits
}

# 3. Setup the tuning pipeline
tuning_pipeline = Pipeline(steps=[('preprocessor', preprocessor), ('classifier', RandomForestClassifier(random_state=42))])

# 4. Run Grid Search
grid_search = GridSearchCV(tuning_pipeline, param_grid, cv=3, scoring='f1', n_jobs=-1)
grid_search.fit(X_train, y_train)

# 5. Get the best tuned model
best_model = grid_search.best_estimator_
y_tuned_pred = best_model.predict(X_test)

# 6. Store results
results['Tuned Random Forest'] = {
    'Accuracy': accuracy_score(y_test, y_tuned_pred),
    'Precision': precision_score(y_test, y_tuned_pred, zero_division=0),
    'Recall': recall_score(y_test, y_tuned_pred, zero_division=0),
    'F1-Score': f1_score(y_test, y_tuned_pred, zero_division=0)
}

# Print metrics to the screen
print(f"Best Grid Parameters: {grid_search.best_params_}")
print("\n[Tuned Random Forest]")
for metric, value in results['Tuned Random Forest'].items():
    print(f"  {metric}: {value:.4f}")

# Plot performance results chart
df_results = pd.DataFrame(results).T
df_results.plot(kind='bar', figsize=(10, 6))
plt.title('Model Performance Comparison on Custom House Dataset')
plt.ylabel('Evaluation Score Metrics')
plt.ylim(0, 1.1)
plt.xticks(rotation=15)
plt.grid(axis='y', linestyle='--', alpha=0.7)
plt.legend(bbox_to_anchor=(1.05, 1), loc='upper left')
plt.tight_layout()
plt.show()
